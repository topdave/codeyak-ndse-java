1   package com.codeyak.wordgame.words;

3   import java.io.BufferedReader;
4   import java.io.File;
5   import java.io.FileReader;
6   import java.io.IOException;
7   import java.util.ArrayList;
8   import java.util.Collections;
9   import java.util.List;
10  
11  import com.codeyak.wordgame.Constants;
12  
13  /**
14   * contains an alphabetically sorted list of valid words
15   * 
16   * @author dave_blake
17   *
18   */
19  public class WordList {
20  
21      List<String> words = new ArrayList<String>();
22      
23      public WordList(File file) {
24          try {
25              load(file);
26          } catch (IOException e) {
27              e.printStackTrace();
28          }
29      }
30  
31      protected void load(File file) throws IOException {
32          BufferedReader br = new BufferedReader(new FileReader(file));
33          while (true) {
34              String line = br.readLine();
35              if (line == null) break;
36              words.add(line.toUpperCase());
37          }
38          Collections.sort(words);
39          br.close();
40      }
41  
42      public int getSize() {
43          return words.size();
44      }
45      
46      public String toString() {
47          StringBuilder sb = new StringBuilder();
48          for (String word : words) {
49              sb.append(word + Constants.NL);
50          }
51          return sb.toString();
52      }
53      
54  
55  }
