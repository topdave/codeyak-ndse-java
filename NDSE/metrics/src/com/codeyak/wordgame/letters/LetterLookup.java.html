13  public abstract class LetterLookup {
14      protected Map<Character, Integer> lookup = new HashMap<Character, Integer>();
15      
16      public List<Character> getSortedLetters() {
17          List<Character> rv = new ArrayList<Character>(lookup.keySet());
18          Collections.sort(rv);
19          return rv;
20      }
21      
22      public String toString() {
23          StringBuilder sb = new StringBuilder();
24          for (Character c : getSortedLetters()) {
25              if (sb.length() > 0) sb.append(", ");
26              sb.append(c+"="+lookup.get(c));
27          }
28          return sb.toString();
29      }
30      
31      protected void add(char c, String v) {
32          lookup.put(c, Integer.parseInt(v));
33      }
34      
35      protected void load(File file) throws IOException {
36          BufferedReader br = new BufferedReader(new FileReader(file));
37          while (true) {
38              String line = br.readLine();
39              if (line == null) break;
40              String[] split = line.split(" ");
41              char c = split[0].charAt(0);
42              int v = Integer.parseInt(split[1]);
43              lookup.put(c, v);
44          }
45          br.close();
46      }
47      
48      public int getTotal() {
49          int rv = 0;
50          for (int i : lookup.values()) {
51              rv += i;
52          }
53          return rv;
54      }
55          
56      public String getLetters() {
57          StringBuilder sb = new StringBuilder();
58          for (char c : getSortedLetters()) {
59              sb.append(c);
60          }
61          return sb.toString();
62      }
63      
64      public int getNumber(char c) {
65          return lookup.get(c);
66      }
67  }
