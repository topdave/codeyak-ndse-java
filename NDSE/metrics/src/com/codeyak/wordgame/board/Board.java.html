26      protected void load(File file) throws IOException {
27          BufferedReader br = new BufferedReader(new FileReader(file));
28          String header = br.readLine();
29          String[] xy = header.split("x");
30          width = Integer.parseInt(xy[0]);
31          height = Integer.parseInt(xy[1]);
32          letterMultipliers = new byte[height][width];
33          wordMultipliers = new byte[height][width];
34          int y = 0;
35          while (true) {
36              String line = br.readLine();
37              if (line == null)
38                  break;
39              if (line.startsWith("+"))
40                  continue;
41              char[] row = line.toCharArray();
42              for (int x = 0; x < width; x++) {
43                  char c = row[x + 1];
44                  byte wm = 1;
45                  byte lm = 1;
46                  switch (c) {
47                  case '2':
48                      wm = 2;
49                      break;
50                  case '3':
51                      wm = 3;
52                      break;
53                  case '4':
54                      wm = 4;
55                      break;
56                  case 'b':
57                      lm = 2;
58                      break;
59                  case 'c':
60                      lm = 3;
61                      break;
62                  case 'd':
63                      lm = 4;
64                      break;
65                  }
66                  letterMultipliers[y][x] = lm;
67                  wordMultipliers[y][x] = wm;
68              }
69              y++;
70          }
71          br.close();
72      }
73  
74      public String toString() {
75          StringBuilder sb = new StringBuilder();
76          sb.append(getTopBorder());
77          for (int y = 0; y < height; y++) {
78              sb.append("|");
79              for (int x = 0; x < width; x++) {
80                  int lm = letterMultipliers[y][x];
81                  int wm = wordMultipliers[y][x];
82                  if (wm > 1)
83                      sb.append(wm);
84                  else if (lm > 1)
85                      sb.append((char)('a' + lm));
86                  else
87                      sb.append(" ");
88              }
89              sb.append("|" + Constants.NL);
90          }
91          sb.append(getTopBorder());
92          return sb.toString();
93      }
94  
95      private String getTopBorder() {
96          StringBuilder sb = new StringBuilder();
97          sb.append("+");
98          for (int x = 0; x < width; x++) {
99              sb.append("-");
100         }
101         sb.append("+"+Constants.NL);
102         return sb.toString();
103     }
