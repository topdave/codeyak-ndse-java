109         int pos = 0;
110         
111         if (firstPlay) {

115             for (int i=0; i<tiles.length; i++) {

117                 //left
118                 if (i % width > 0 && tiles[i-1] > 0) {

124                     potentialHooksReusable[pos++] = i;
125                     continue;
126                 }
127                 //above
128                 if (i / width > 0 && tiles[i-width] > 0) {

131                 }
